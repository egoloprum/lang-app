#include <iostream>
#include <stdlib.h>
#include <time.h>   
using namespace std;

int main() {
	srand(time(NULL));
	int H[30];

	for (int i = 0; i < 30; i++) {
		H[i] = rand();
		cout << H[i] << " ";
	}

	int max = -1000;

	for (int i = 0; i < 30; i++) {
		if (H[i] > max) {
			max = H[i];
		}
	}
	cout << endl << max << endl;

	for (int i = 0; i < 30; i++) {
		if (H[i] == max) {
			H[i] = NULL;
		}
	}

	for (int i = 0; i < 30; i++) {
		cout << H[i] << " ";
	}

}
